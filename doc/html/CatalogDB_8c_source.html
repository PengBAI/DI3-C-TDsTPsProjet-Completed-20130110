00066 /** Create a new string on the heap containing the value of the specified field for the specified record of a database
00067  * @param catalogDB the database
00068  * @param recordIndex the record index
00069  * @param field the field to query
00070  * @return a new string
00071  * @note The string is allocated using malloc().
00072  * @warning the user is responsible for freeing the memory allocated for the new string
00073  */
00074 char * CatalogDB_getFieldValueAsString(CatalogDB * catalogDB, int recordIndex, int field) {
00075     char * content = NULL;
00076     if (catalogDB != NULL) {
00077         CatalogRecord_FieldProperties properties = CatalogRecord_getFieldProperties(field);
00078         CatalogRecord record;
00079         CatalogRecord_init(&record);
00080         CatalogDB_readRecord(catalogDB, recordIndex, &record);
00081         content = properties.getValue(&record);
00082         CatalogRecord_finalize(&record);
00083     }
00084     return content;
00085 }
