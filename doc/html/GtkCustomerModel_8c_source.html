00213 /*****************************************************************************
00214  *
00215  *  GtkCustomerModel_get_iter: converts a tree path (physical position) into a
00216  *                        tree iter structure (the content of the iter
00217  *                        fields will only be used internally by our model).
00218  *                        We simply store a pointer to our CustomRecord
00219  *                        structure that represents that row in the tree iter.
00220  *
00221  *****************************************************************************/
00222 
00223 static gboolean GtkCustomerModel_get_iter(GtkTreeModel *tree_model,
00224         GtkTreeIter *iter, GtkTreePath *path) {
00225     GtkCustomerModel *custom_list;
00226     gint *indices, n, depth;
00227 
00228     g_assert(GTKCUSTOMERMODEL_IS_LIST(tree_model));
00229     g_assert(path!=NULL);
00230 
00231     custom_list = GTKCUSTOMERMODEL(tree_model);
00232 
00233     indices = gtk_tree_path_get_indices(path);
00234     depth = gtk_tree_path_get_depth(path);
00235 
00236     /* we do not allow children */
00237     g_assert(depth == 1); /* depth 1 = top level; a list only has top level nodes and no children */
00238 
00239     n = indices[0]; /* the n-th top level row */
00240 
00241     if (n >= CustomerDB_getRecordCount(custom_list->clientDB) || n < 0)
00242         return FALSE;
00243 
00244     /* We simply store a pointer to our custom record in the iter */
00245     iter->stamp = custom_list->stamp;
00246     iter->user_data = GINT_TO_POINTER(n);
00247     iter->user_data2 = NULL; /* unused */
00248     iter->user_data3 = NULL; /* unused */
00249 
00250     return TRUE;
00251 }
