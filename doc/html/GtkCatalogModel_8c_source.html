00200 /*****************************************************************************
00201  *
00202  *  GtkCatalogModel_get_iter: converts a tree path (physical position) into a
00203  *                        tree iter structure (the content of the iter
00204  *                        fields will only be used internally by our model).
00205  *                        We simply store a pointer to our CustomRecord
00206  *                        structure that represents that row in the tree iter.
00207  *
00208  *****************************************************************************/
00209 
00210 static gboolean GtkCatalogModel_get_iter(GtkTreeModel *tree_model, GtkTreeIter *iter,
00211         GtkTreePath *path) {
00212     GtkCatalogModel *custom_list;
00213     gint *indices, n, depth;
00214 
00215     g_assert(GTKCATALOGMODEL_IS_LIST(tree_model));
00216     g_assert(path!=NULL);
00217 
00218     custom_list = GTKCATALOGMODEL(tree_model);
00219 
00220     indices = gtk_tree_path_get_indices(path);
00221     depth = gtk_tree_path_get_depth(path);
00222 
00223     /* we do not allow children */
00224     g_assert(depth == 1); /* depth 1 = top level; a list only has top level nodes and no children */
00225 
00226     n = indices[0]; /* the n-th top level row */
00227 
00228     if (n >= CatalogDB_getRecordCount(custom_list->catalogDB) || n < 0)
00229         return FALSE;
00230 
00231     /* We simply store a pointer to our custom record in the iter */
00232     iter->stamp = custom_list->stamp;
00233     iter->user_data = GINT_TO_POINTER(n);
00234     iter->user_data2 = NULL; /* unused */
00235     iter->user_data3 = NULL; /* unused */
00236 
00237     return TRUE;
00238 }
